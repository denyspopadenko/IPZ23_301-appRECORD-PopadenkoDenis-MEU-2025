ЗАВДАННЯ №2

Мета: Створити функції для перевірки на спам та розрахунку покупки з перевіркою балансу.


// 1. Перевірка на спам
function checkForSpam(message) {
    const lowerMessage = message.toLowerCase();
    return lowerMessage.includes("spam") || lowerMessage.includes("sale");
}

// 2. Покупка дроїдів з перевіркою кредитів
function makeTransaction(quantity, pricePerDroid, customerCredits) {
    const totalPrice = quantity * pricePerDroid;
    if (totalPrice > customerCredits) {
        return "Insufficient funds!";
    } else {
        return `You ordered ${quantity} droids worth ${totalPrice} credits!`;
    }
}

// Перевірка
console.log(checkForSpam("Latest Technology News")); // false
console.log(checkForSpam("Get best SALE offers now!")); // true
console.log(checkForSpam("Amazing SalE, only tonight!")); // true

console.log(makeTransaction(5, 3000, 23000)); // Success
console.log(makeTransaction(8, 2000, 10000)); // Insufficient

Результат виконання (Console):

> false
> true
> true
> "You ordered 5 droids worth 15000 credits!"
> "Insufficient funds!"

ЗАВДАННЯ №3

Мета: Робота з методами масивів (фільтрація та об'єднання).


// 1. Фільтрація масиву
function filterArray(numbers, value) {
    const filteredNumbers = [];
    for (const number of numbers) {
        if (number > value) {
            filteredNumbers.push(number);
        }
    }
    return filteredNumbers;
}

// 2. Створення нового масиву з обмеженням довжини
function makeArray(firstArray, secondArray, maxLength) {
    const newArray = firstArray.concat(secondArray);
    if (newArray.length > maxLength) {
        return newArray.slice(0, maxLength);
    }
    return newArray;
}

// Перевірка
console.log(filterArray([1, 2, 3, 4, 5], 3)); // [4, 5]
console.log(filterArray([12, 24, 8, 41, 76], 20)); // [24, 41, 76]

console.log(makeArray(["Mango", "Poly"], ["Ajax", "Chelsea"], 3)); 
// ["Mango", "Poly", "Ajax"]
console.log(makeArray(["Mango", "Poly"], ["Ajax", "Chelsea"], 5)); 
// ["Mango", "Poly", "Ajax", "Chelsea"]

Результат виконання (Console):

> [4, 5]
> [24, 41, 76]
> ["Mango", "Poly", "Ajax"]
> ["Mango", "Poly", "Ajax", "Chelsea"]

ЗАВДАННЯ №4 (Варіант 8)

Завдання 1: Знайти найменший елемент першого масиву, який також є у другому масиві.

Завдання 2: Сортування бульбашкою.


// --- Підзавдання 1: Пошук спільного мінімуму ---
function findMinCommon(arr1, arr2) {
    let minCommon = null;
    
    for (let num of arr1) {
        // Якщо число є в другому масиві
        if (arr2.includes(num)) {
            if (minCommon === null || num < minCommon) {
                minCommon = num;
            }
        }
    }
    return minCommon;
}

const arrayA = [15, 42, 8, 23, 4];
const arrayB = [8, 55, 4, 23, 90];
console.log("Масив 1:", arrayA);
console.log("Масив 2:", arrayB);
console.log("Найменший спільний:", findMinCommon(arrayA, arrayB)); // 4

// --- Підзавдання 2: Сортування бульбашкою ---
function bubbleSort(arr) {
    let len = arr.length;
    // Копіюємо масив, щоб не мутувати вихідний
    let sortedArr = [...arr]; 
    
    for (let i = 0; i < len; i++) {
        for (let j = 0; j < len - i - 1; j++) {
            if (sortedArr[j] > sortedArr[j + 1]) {
                let temp = sortedArr[j];
                sortedArr[j] = sortedArr[j + 1];
                sortedArr[j + 1] = temp;
            }
        }
    }
    return sortedArr;
}

const inputArr = [64, 34, 25, 12, 22, 11, 90];
console.log("Вхідний масив:", inputArr);
console.log("Відсортований масив:", bubbleSort(inputArr));

Результат виконання (Console):

> "Найменший спільний:" 4
--------------------------------

> "Відсортований масив:" [11, 12, 22, 25, 34, 64, 90]

// Логіка роботи будильника
function setAlarm() {
    const inputTime = document.getElementById('alarm-time').value;
    const now = new Date();
    const alarm = new Date();
    const [hours, minutes] = inputTime.split(':');
    
    alarm.setHours(hours);
    alarm.setMinutes(minutes);
    alarm.setSeconds(0);

    // Розрахунок часу до спрацювання
    const timeToAlarm = alarm - now;
    
    console.log(`Alarm set for ${hours}:${minutes}`);

    // Таймер спрацювання
    setTimeout(() => {
        alert("⏰ ЧАС ПРОКИДАТИСЯ! ⏰");
        console.log("Alarm Triggered!");
    }, timeToAlarm);
}
