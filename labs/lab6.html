Завдання №2

Умова: Напишіть функцію getProductDetails, яка приймає ідентифікатор товару та дві колбек-функції (успіх та помилка).


const products = [
    { id: 1, name: "Laptop", price: 20000 },
    { id: 2, name: "Phone", price: 15000 },
    { id: 3, name: "Tablet", price: 8000 }
];

function getProductDetails(productId, successCallback, errorCallback) {
    const product = products.find(p => p.id === productId);
    if (product) {
        successCallback(product);
    } else {
        errorCallback(`Error: Product with ID ${productId} not found.`);
    }
}

// Виклики
getProductDetails(2, 
    (p) => console.log("Success:", p), 
    (e) => console.log(e)
);

getProductDetails(99, 
    (p) => console.log("Success:", p), 
    (e) => console.log(e)
);

Результат виконання:

> Success: { id: 2, name: "Phone", price: 15000 }
> Error: Product with ID 99 not found.

Умова: З об'єкту концертів отримати масив міст, де концерт ще не пройшов, та відсортувати їх хронологічно.


const concerts = {
    Київ: new Date("2020-04-01"),
    Умань: new Date("2025-07-02"),
    Вінниця: new Date("2020-04-21"),
    Одеса: new Date("2025-03-15"),
    Хмельницький: new Date("2020-04-18"),
    Харків: new Date("2025-07-10"),
};

const processConcerts = (concertsObj) => {
    const now = new Date();
    return Object.entries(concertsObj)
        .filter(([city, date]) => date > now) // Лише майбутні
        .sort((a, b) => a[1] - b[1])          // Сортування за датою
        .map(([city]) => city);               // Лише назви міст
};

console.log(processConcerts(concerts));

Результат виконання:

> ["Одеса", "Умань", "Харків"]

const medicines = [
    { name: "Noshpa", price: 170 },
    { name: "Analgin", price: 55 },
    { name: "Quanil", price: 310 },
    { name: "Alphacholine", price: 390 },
];

const applyDiscount = (items) => {
    return items.map((item, index) => {
        const newItem = { ...item, id: index + 1 };
        if (newItem.price > 300) {
            newItem.price *= 0.7; // Знижка 30%
        }
        return newItem;
    });
};

console.log(applyDiscount(medicines));

Завдання №8

Умова: Написати функцію-конструктор Storage з методами getItems, addItems, removeItem.


function Storage(items) {
    this.items = items;

    this.getItems = function() {
        return this.items;
    };
    this.addItems = function(item) {
        this.items.push(item);
    };
    this.removeItem = function(item) {
        const idx = this.items.indexOf(item);
        if (idx !== -1) this.items.splice(idx, 1);
    };
}

const storage = new Storage(["apple", "banana", "mango"]);
console.log(storage.getItems());

storage.addItems("orange");
console.log("Added orange:", storage.getItems());

storage.removeItem("banana");
console.log("Removed banana:", storage.getItems());

Результат виконання:

> ["apple", "banana", "mango"]
> "Added orange:" ["apple", "banana", "mango", "orange"]
> "Removed banana:" ["apple", "mango", "orange"]

Умова: Порахувати кількість кожного тегу в масиві твітів.


const tweets = [
    { id: "000", likes: 5, tags: ["js", "nodejs"] },
    { id: "001", likes: 2, tags: ["html", "css"] },
    { id: "002", likes: 17, tags: ["html", "js", "nodejs"] },
    { id: "003", likes: 8, tags: ["css", "react"] },
    { id: "004", likes: 0, tags: ["js", "nodejs", "react"] },
];

const tagsStats = tweets
    .flatMap(t => t.tags) // Збираємо всі теги в один масив
    .reduce((acc, tag) => {
        acc[tag] = (acc[tag] || 0) + 1;
        return acc;
    }, {});

console.log(tagsStats);

> { js: 3, nodejs: 3, html: 2, css: 2, react: 2 }

Завдання №10

Умова: Написати функцію checkBrackets(str) для перевірки правильності закриття дужок.


function checkBrackets(str) {
    const stack = [];
    const pairs = { '(': ')', '{': '}', '[': ']' };

    for (const char of str) {
        if (pairs[char]) {
            stack.push(char);
        } else if (Object.values(pairs).includes(char)) {
            if (pairs[stack.pop()] !== char) return false;
        }
    }
    return stack.length === 0;
}

console.log(checkBrackets("function f() { return [1]; }")); // true
console.log(checkBrackets("{ [ } ]")); // false

Результат виконання:

> true
> false
